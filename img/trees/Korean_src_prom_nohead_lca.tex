% MG lexical item

%	argument 1: phonetic exponent

% Labeling Macros
% Format:
%	argument 1 = node label
%	argument 2 = step at which node is put into memory
% 	argument 3 = step at which node is flushed from memory
% Example:
%	Merge node, not held in memory: \Lab{Merge}{5}{6}
